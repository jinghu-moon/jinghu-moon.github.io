为你的博客添加书签页

博主还没有学习到 JS，有些功能还不能实现。比如给每个网址自动添加 logo。等到我学到了 JS，我会尝试实现该功能。

1.1 新建书签页

右键打开 Git Bash Here ，新建页面。

hexo n page bookmarks

1.2 JS 文件

在 Blog\source\js 路径，新建 bookmarks.js 文件。添加如下代码。

// 采用一个多对象JSON文件存储所有数据的方式
stars = {
  init: function (url) {
    var that = this
    $.getJSON(url, function (data) {
      for (var className in data) {
        var classData = data[className]
        that.render(classData, className)
      }
    })
  },
  render: function (data, name) {
    var nickname,
      site,
      li = ''
    for (var i = 0; i < data.length; i++) {
      nickname = data[i].nickname
      site = data[i].site
      content = data[i].content
      li +=
        '<div class="card" onclick="window.open(\'' +
        site +
        '\')" >' +
        '<div class="card-header">' +
        '<div>' +
        nickname +
        '</div>' +
        '</div>' +
        '<div class="card-content">' +
        '<div>' +
        content +
        '</div>' +
        '</div>' +
        '</div>'
    }
    $(name).append(li)
  },
}

// 传入json文件的路径
stars.init('../allbookmarks.json')

custom_js:
  - /js/bookmarks.js

menu:
  - { key: "书签", link: "/stars/", icon: "iconfont icon-shoucang" }
